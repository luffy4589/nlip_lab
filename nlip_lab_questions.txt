  Building an N-gram LM from scratch
pythonfrom collections import defaultdict, Counter
import math

class NgramLM:
    def __init__(self, n):
        self.n = n
        self.ngram_counts  = defaultdict(Counter)
        self.context_counts = defaultdict(int)

    def train(self, sentences):
        for sent in sentences:
            tokens = ['<s>'] * (self.n - 1) + sent.split() + ['</s>']
            for i in range(self.n - 1, len(tokens)):
                context = tuple(tokens[i - self.n + 1 : i])
                word    = tokens[i]
                self.ngram_counts[context][word] += 1
                self.context_counts[context]     += 1

    def prob(self, word, context):
        context = tuple(context[-(self.n-1):])
        count   = self.ngram_counts[context][word]
        total   = self.context_counts[context]
        return count / total if total > 0 else 0.0

    def score_sentence(self, sentence):
        tokens = ['<s>'] * (self.n - 1) + sentence.split() + ['</s>']
        log_prob = 0.0
        for i in range(self.n - 1, len(tokens)):
            context = tokens[i - self.n + 1 : i]
            p = self.prob(tokens[i], context)
            log_prob += math.log(p) if p > 0 else float('-inf')
        return log_prob

corpus = [
    "the cat sat on the mat",
    "the dog lay on the rug",
    "the cat lay on the rug",
]
lm = NgramLM(n=3)
lm.train(corpus)
print(lm.score_sentence("the cat sat"))   # log probability

Smoothing — the critical problem
Raw N-gram counts fail badly on unseen N-grams (probability = 0 kills the whole sentence). Smoothing redistributes probability mass to unseen events.
Laplace (add-k) smoothing
pythonclass LaplaceLM(NgramLM):
    def __init__(self, n, vocab_size, k=1):
        super().__init__(n)
        self.vocab_size = vocab_size
        self.k = k

    def prob(self, word, context):
        context = tuple(context[-(self.n-1):])
        count = self.ngram_counts[context][word]
        total = self.context_counts[context]
        return (count + self.k) / (total + self.k * self.vocab_size)
Kneser-Ney smoothing (best in practice)
Kneser-Ney is the gold standard for MT language models. It uses a discount d and a continuation probability — how many unique contexts a word appears in, not just how often.
pythonclass KneserNeyLM:
    def __init__(self, n, d=0.75):
        self.n   = n
        self.d   = d
        self.ngram_counts   = defaultdict(Counter)
        self.context_counts = defaultdict(int)
        self.continuation   = Counter()   # how many contexts each word follows
        self.num_bigrams    = 0

    def train(self, sentences):
        for sent in sentences:
            tokens = ['<s>'] * (self.n - 1) + sent.split() + ['</s>']
            for i in range(1, len(tokens)):
                self.continuation[tokens[i]] += 1
                self.num_bigrams += 1
            for i in range(self.n - 1, len(tokens)):
                ctx  = tuple(tokens[i - self.n + 1 : i])
                word = tokens[i]
                self.ngram_counts[ctx][word] += 1
                self.context_counts[ctx]     += 1

    def p_continuation(self, word):
        return self.continuation[word] / self.num_bigrams if self.num_bigrams else 0

    def prob(self, word, context):
        ctx   = tuple(context[-(self.n-1):])
        count = self.ngram_counts[ctx].get(word, 0)
        total = self.context_counts[ctx]
        if total == 0:
            return self.p_continuation(word)
        # Number of unique words seen after this context
        lambda_ctx = (self.d / total) * len(self.ngram_counts[ctx])
        discounted = max(count - self.d, 0) / total
        return discounted + lambda_ctx * self.p_continuation(word)
Interpolation (Jelinek-Mercer)
Blend higher- and lower-order models. When you haven't seen "sat on the mat" (trigram), fall back to "on the mat" (bigram), then "the mat" (unigram):
pythonclass InterpolatedLM:
    def __init__(self, models, lambdas):
        assert abs(sum(lambdas) - 1.0) < 1e-9
        self.models  = models    # [trigram_lm, bigram_lm, unigram_lm]
        self.lambdas = lambdas   # [0.6, 0.3, 0.1]

    def prob(self, word, context):
        return sum(lam * m.prob(word, context)
                   for lam, m in zip(self.lambdas, self.models))

Using N-gram LMs for decoding in MT
The language model scores candidate translations during beam search. Each hypothesis is extended one word at a time, scored by P(translation_model) * P(language_model).
pythonimport heapq

def beam_search(translation_options, lm, beam_size=3, n=3):
    """
    translation_options: list of (source_phrase, [candidate_translations])
    Returns the best translation hypothesis.
    """
    # Each state: (neg_log_prob, tokens_so_far)
    beam = [(0.0, ['<s>'] * (n - 1))]

    for src_phrase, candidates in translation_options:
        new_beam = []
        for neg_logp, tokens in beam:
            for cand in candidates:
                cand_tokens = cand.split()
                score = neg_logp
                ctx   = list(tokens)
                for word in cand_tokens:
                    p = lm.prob(word, ctx[-(n-1):])
                    score -= math.log(p + 1e-10)
                    ctx.append(word)
                new_beam.append((score, ctx))
        # Keep only top beam_size hypotheses
        beam = heapq.nsmallest(beam_size, new_beam, key=lambda x: x[0])

    best_score, best_tokens = beam[0]
    return ' '.join(best_tokens[n-1:])   # strip <s> padding

Evaluation: Perplexity and BLEU
Perplexity measures how well the LM predicts a held-out test set. Lower = better.
pythondef perplexity(lm, test_sentences):
    total_log_prob = 0.0
    total_words    = 0
    for sent in test_sentences:
        tokens = ['<s>'] * (lm.n - 1) + sent.split() + ['</s>']
        for i in range(lm.n - 1, len(tokens)):
            ctx = tokens[i - lm.n + 1 : i]
            p   = lm.prob(tokens[i], ctx)
            total_log_prob += math.log(p + 1e-10)
            total_words    += 1
    return math.exp(-total_log_prob / total_words)

print(f"Perplexity: {perplexity(lm, test_sentences):.2f}")

pythonfrom scipy import signal
import numpy as np

fs = 44100       # sample rate (Hz)
fc = 1000        # cutoff frequency (Hz)
order = 4        # filter order — higher = steeper rolloff

# Normalize cutoff to Nyquist (0 to 1)
nyq = fs / 2
wn  = fc / nyq

# --- Low-pass filter ---
b_lp, a_lp = signal.butter(order, wn, btype='low')

# --- High-pass filter ---
b_hp, a_hp = signal.butter(order, wn, btype='high')

# Apply to a signal
t = np.linspace(0, 1, fs)
x = np.sin(2*np.pi*200*t) + np.sin(2*np.pi*5000*t)   # 200Hz + 5kHz mix

y_lp = signal.lfilter(b_lp, a_lp, x)   # keeps low tones
y_hp = signal.lfilter(b_hp, a_hp, x)   # keeps high tones
2. Zero-phase filtering (no phase distortion)
lfilter shifts the signal in time. Use filtfilt when phase matters:
pythony_lp_zp = signal.filtfilt(b_lp, a_lp, x)   # applies filter forward + backward
y_hp_zp = signal.filtfilt(b_hp, a_hp, x)   # result: zero phase lag
3. Choosing the right filter design
python# Butterworth — maximally flat passband (most common)
b, a = signal.butter(4, wn, btype='low')

# Chebyshev Type I — steeper rolloff, ripple in passband
b, a = signal.cheby1(4, rp=1, Wn=wn, btype='low')   # rp = max ripple dB

# Chebyshev Type II — steeper rolloff, ripple in stopband
b, a = signal.cheby2(4, rs=40, Wn=wn, btype='low')  # rs = stopband atten. dB

# Elliptic — steepest rolloff, ripple in both bands
b, a = signal.ellip(4, rp=1, rs=40, Wn=wn, btype='low')

# Bessel — linear phase (preserves waveform shape), gentler rolloff
b, a = signal.bessel(4, wn, btype='low')

5. Simple 1st-order filter from scratch (exponential moving average)
This is the discrete-time implementation of a first-order RC filter — dead simple, used everywhere in audio and control systems:
pythonimport numpy as np

def lowpass_ema(x, fc, fs):
    """First-order low-pass via exponential moving average."""
    dt   = 1.0 / fs
    tau  = 1.0 / (2 * np.pi * fc)
    alpha = dt / (tau + dt)          # α = dt / (RC + dt)
    y = np.zeros_like(x)
    y[0] = x[0]
    for i in range(1, len(x)):
        y[i] = alpha * x[i] + (1 - alpha) * y[i-1]
    return y

def highpass_ema(x, fc, fs):
    """First-order high-pass via complementary of low-pass."""
    return x - lowpass_ema(x, fc, fs)

fs = 44100
t  = np.linspace(0, 0.05, int(0.05 * fs))
x  = np.sin(2*np.pi*200*t) + np.sin(2*np.pi*5000*t)

y_lp = lowpass_ema(x, fc=1000, fs=fs)
y_hp = highpass_ema(x, fc=1000, fs=fs)

7. Real-time / streaming filter (IIR direct form)
For audio plugins, control loops, or any live signal:
pythonclass BiquadFilter:
    """Direct Form II transposed biquad — numerically stable for real-time use."""
    def __init__(self, b, a):
        self.b = b / a[0]
        self.a = a / a[0]
        self.z = [0.0, 0.0]   # state registers

    def process_sample(self, x):
        y       = self.b[0]*x + self.z[0]
        self.z[0] = self.b[1]*x - self.a[1]*y + self.z[1]
        self.z[1] = self.b[2]*x - self.a[2]*y
        return y

    def process_block(self, block):
        return np.array([self.process_sample(s) for s in block])

def load_images_from_folder(root_dir, img_size=(64, 64), grayscale=False):
    """
    Expects folder structure:
      root_dir/
        class_a/  img1.jpg  img2.jpg ...

                img = cv2.cvtColor(img, cv2.COLOR_BGR2RGB)
            X.append(img.flatten())
            y.append(label_idx)

6. Histogram statistics
pythonimport numpy as np

def histogram_stats(image_gray):
    hist = cv2.calcHist([image_gray], [0], None, [256], [0, 256]).flatten()
    total = hist.sum()
    intensities = np.arange(256)

    mean   = np.sum(intensities * hist) / total
    var    = np.sum(((intensities - mean) ** 2) * hist) / total
    std    = np.sqrt(var)
    median = np.searchsorted(np.cumsum(hist), total / 2)
    mode   = np.argmax(hist)

    # Skewness: negative = left-skewed (dark image), positive = right-skewed (bright)
    skew = np.sum(((intensities - mean) ** 3) * hist) / (total * std ** 3)

    # Kurtosis: low = flat distribution, high = peaked
    kurt = np.sum(((intensities - mean) ** 4) * hist) / (total * std ** 4) - 3

    return {
        "mean":     round(mean, 2),
        "std":      round(std, 2),
        "median":   int(median),
        "mode":     int(mode),
        "skewness": round(skew, 3),
        "kurtosis": round(kurt, 3),
    }

stats = histogram_stats(gray)
for k, v in stats.items():
    print(f"  {k:12s}: {v}")

pythondef compare_images(img1_path, img2_path):
    img1 = cv2.imread(img1_path)
    img2 = cv2.imread(img2_path)

    # Convert to HSV for color comparison
    h1 = cv2.calcHist([cv2.cvtColor(img1, cv2.COLOR_BGR2HSV)],
                      [0, 1], None, [50, 60], [0, 180, 0, 256])
    h2 = cv2.calcHist([cv2.cvtColor(img2, cv2.COLOR_BGR2HSV)],

    # Four comparison methods:
    methods = {

        "Chi-squared":        cv2.HISTCMP_CHISQR,        # lower  = more similar
        "Intersection":       cv2.HISTCMP_INTERSECT,     # higher = more similar
        "Bhattacharyya dist": cv2.HISTCMP_BHATTACHARYYA, # lower  = more similar
    }
    for name, method in methods.items():
        score = cv2.compareHist(h1, h2, method)
        print(f"  {name:20s}: {score:.4f}")

1. Color quantization — K-Means on a single image
The most classic use: reduce an image from millions of colors down to K dominant colors, creating a posterized effect.
pythonimport cv2
import numpy as np
import matplotlib.pyplot as plt

def color_quantization(image_path, k=8):
    img = cv2.imread(image_path)
    img_rgb = cv2.cvtColor(img, cv2.COLOR_BGR2RGB)

    # Reshape to (num_pixels, 3) — each row is one RGB pixel
    pixels = img_rgb.reshape(-1, 3).astype(np.float32)

    # K-Means
    criteria = (cv2.TERM_CRITERIA_EPS + cv2.TERM_CRITERIA_MAX_ITER, 100, 0.2)
    _, labels, centers = cv2.kmeans(
        pixels, k, None, criteria,
        attempts=10,
        flags=cv2.KMEANS_PP_CENTERS   # K-Means++ initialization
    )

    centers = np.uint8(centers)
    # Replace each pixel with its cluster center color
    quantized = centers[labels.flatten()].reshape(img_rgb.shape)

    # Show dominant color palette
    palette = np.zeros((60, k * 60, 3), dtype=np.uint8)
    for i, color in enumerate(centers):
        palette[:, i*60:(i+1)*60] = color

    fig, axes = plt.subplots(1, 3, figsize=(13, 4))
    axes[0].imshow(img_rgb);    axes[0].set_title("Original");      axes[0].axis("off")
    axes[1].imshow(quantized);  axes[1].set_title(f"Quantized (K={k})"); axes[1].axis("off")

3. Better features: HOG + PCA before K-Means
Raw pixel vectors are high-dimensional and noisy. HOG + PCA gives much better clusters.
pythonfrom skimage.feature import hog
from sklearn.decomposition import PCA
from sklearn.cluster import KMeans

def extract_hog(image_path, img_size=(128, 128)):
    img = cv2.imread(image_path)
    img = cv2.resize(img, img_size)
    gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
    features = hog(
        gray,
        orientations=9,
        pixels_per_cell=(8, 8),
        cells_per_block=(2, 2),
        block_norm='L2-Hys'
    )
    return features

4. CNN features + K-Means (most powerful)
Use a pretrained ResNet to extract rich semantic embeddings, then cluster those.
pythonimport torch
import torchvision.models as models
import torchvision.transforms as transforms
from PIL import Image
from sklearn.cluster import KMeans
import numpy as np

device = 'cuda' if torch.cuda.is_available() else 'cpu'

# Load ResNet-18, strip the final classification head
model = models.resnet18(weights='IMAGENET1K_V1')
model.fc = torch.nn.Identity()   # output: 512-dim embedding
model = model.to(device).eval()

transform = transforms.Compose([
    transforms.Resize((224, 224)),
    transforms.ToTensor(),
    transforms.Normalize(mean=[0.485, 0.456, 0.406],
                         std =[0.229, 0.224, 0.225])
])

def extract_embeddings(image_paths, batch_size=32):
    all_embeds = []
    with torch.no_grad():
        for i in range(0, len(image_paths), batch_size):
            batch_paths = image_paths[i:i+batch_size]
            tensors = []
            for p in batch_paths:
                img = Image.open(p).convert('RGB')
                tensors.append(transform(img))
            batch = torch.stack(tensors).to(device)
            embeds = model(batch).cpu().numpy()
            all_embeds.append(embeds)
    return np.vstack(all_embeds)

embeddings = extract_embeddings(image_paths)
print(f"Embeddings: {embeddings.shape}")   # (N, 512)

# Optional: PCA before K-Means to speed things up
from sklearn.decomposition import PCA
pca = PCA(n_components=128, random_state=42)
embeddings_pca = pca.fit_transform(embeddings)

kmeans = KMeans(n_clusters=10, init='k-means++', n_init=10, random_state=42)
labels = kmeans.fit_predict(embeddings_pca)

7. Pixel-level segmentation within a single image
K-Means can also segment regions inside one image by clustering pixels by color.
pythonimport cv2
import numpy as np
import matplotlib.pyplot as plt

def segment_image(image_path, k=4):
    img = cv2.imread(image_path)
    img_rgb = cv2.cvtColor(img, cv2.COLOR_BGR2RGB)
    h, w = img_rgb.shape[:2]

    # Feature: pixel RGB + position (gives spatially coherent segments)
    y_coords, x_coords = np.mgrid[0:h, 0:w]
    pixels = np.column_stack([
        img_rgb.reshape(-1, 3).astype(np.float32) / 255.0,
        x_coords.flatten() / w,   # normalize position to [0,1]
        y_coords.flatten() / h,
    ])

    criteria = (cv2.TERM_CRITERIA_EPS + cv2.TERM_CRITERIA_MAX_ITER, 100, 0.2)
    _, labels, centers = cv2.kmeans(pixels, k, None, criteria, 10,
                                     cv2.KMEANS_PP_CENTERS)

    # Render each segment with a distinct color
    colors = (np.array([[255,100,100],[100,200,100],[100,100,255],
                         [255,200,50],[180,100,200],[100,220,220]]) / 255.0)[:k]
    segmented = colors[labels.flatten()].reshape(h, w, 3)

    fig, axes = plt.subplots(1, 2, figsize=(10, 4))
    axes[0].imshow(img_rgb);   axes[0].set_title("Original"); axes[0].axis("off")
    axes[1].imshow(segmented); axes[1].set_title(f"Segmented (K={k})"); axes[1].axis("off")
    plt.tight_layout()
    plt.show()
    return labels.reshape(h, w)

segment_image("photo.jpg", k=5)
